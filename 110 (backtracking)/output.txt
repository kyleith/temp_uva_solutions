program sort(input,output);
var
a,b : integer;
begin
  readln(a,b);
  if a < b then
    writeln(b,a)
  else
    writeln(b,a)
end.

var
a,b,c : integer;
begin
  readln(a,b,c);
  if a < b then
    if a < c then
      writeln(b,c,a)
    if b < c then
      writeln(c,b,a)
    else
      writeln(c,b,a)
  else
    if a < c then
      writeln(b,c,a)
    if b < c then
      writeln(c,b,a)
    else
      writeln(c,b,a)
end.
